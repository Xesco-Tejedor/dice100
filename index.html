    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const dieContainer = document.querySelector('.die-container');

            const die = document.querySelector('.die');

            const dieFace = document.querySelector('.die-face');

            const rollButton = document.getElementById('rollButton');

            const resultText = document.getElementById('resultText');

            let isRolling = false;

            rollButton.addEventListener('click', () => {

                if (isRolling) return;

                isRolling = true;

                rollButton.disabled = true;

                resultText.textContent = "Lanzando...";

                // Resetear estados visuales

                dieContainer.style.transform = 'scale(1)';

                die.style.transform = 'rotate(0deg)';

                dieFace.style.opacity = '0'; // Ocultar número mientras gira

                // Fase 1: Rotación rápida y cambio de números

                let spinCount = 0;

                const maxSpins = 15 + Math.floor(Math.random() * 10); // Entre 15 y 25 cambios rápidos

                let currentRotation = 0;

                const rotationIncrement = 360 / maxSpins * (Math.random() > 0.5 ? 1 : -1) * (2 + Math.random()); // Grados por tick, 2-3 vueltas completas

                const spinInterval = setInterval(() => {

                    // Cambiar el número que se muestra rápidamente

                    dieFace.textContent = Math.floor(Math.random() * 100) + 1;

                    dieFace.style.opacity = '1'; // Mostrar número

                   

                    // Rotar el dado

                    currentRotation += rotationIncrement;

                    die.style.transform = `rotate(${currentRotation}deg)`;

                   

                    spinCount++;

                    if (spinCount >= maxSpins) {

                        clearInterval(spinInterval);

                        finishRoll();

                    }

                }, 100); // Intervalo rápido para el efecto de giro

            });

            function finishRoll() {

                const winningNumber = Math.floor(Math.random() * 100) + 1;

                // Asegurar que la rotación final no sea demasiado loca

                const finalRotation = (Math.floor(Math.random() * 2) + 3) * 360 + (Math.random() * 90 - 45); // 3-4 vueltas + un poco más

                die.style.transform = `rotate(${finalRotation}deg)`;

               

                // Esperar a que la animación de rotación principal termine

                setTimeout(() => {

                    dieFace.textContent = winningNumber;

                    resultText.textContent = `Resultado: ${winningNumber}`;

                   

                    // Fase 2: Acercar la cámara (escalar el contenedor)

                    dieContainer.style.transform = 'scale(1.8)'; // Ajusta el valor de zoom según necesites

                    // Fase 3: Habilitar el botón después de un momento

                    setTimeout(() => {

                        isRolling = false;

                        rollButton.disabled = false;

                    }, 1000); // Tiempo para apreciar el zoom

                   

                }, 2000); // Corresponde a la duración de la transición de 'die' en CSS

            }

        });
